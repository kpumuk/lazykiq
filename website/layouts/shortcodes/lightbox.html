{{- $src := .Get "src" -}} {{- if not $src -}} {{- errorf "lightbox shortcode: src is required" -}} {{- end -}} {{- $alt := .Get "alt" | default "" -}} {{- $class := .Get "class" | default "" -}} {{- $thumbClass := .Get "thumbClass" | default "" -}} {{- $close := .Get "close" | default (printf "#%s-close" (urlize (.Get "alt"))) -}} {{- $id := .Get "id" | default (urlize (.Get "alt")) -}} {{- partial "lightbox.html" (dict "src" $src "alt" $alt "class" $class "thumbClass" $thumbClass "close" $close "id" $id "page" $.Page ) -}}
